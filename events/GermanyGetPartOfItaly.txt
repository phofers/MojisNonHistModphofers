add_namespace = GermanyStealSouthTyrol
	country_event = {

    	id = GermanyStealSouthTyrol.1
    	title = GermanyStealSouthTyrol.1.t
	    desc = GermanyStealSouthTyrol.1.d

		fire_only_once = yes

		mean_time_to_happen = { days = 1 }

	    picture = GFX_report_event_generic_diplomatic
		trigger = {
			OR = {
				AND = {
					country_exists = GER
					GER = {
						OR = {
							has_government = fascism
							has_government = neutrality
						}
					}
				}
				AND = {
					country_exists = AUS
					AUS = {
						OR = {
							has_government = fascism
							has_government = neutrality
						}
					}
				}
    		}

			has_civil_war = yes
			controls_state = 39
			controls_state = 850
			
			OR = {
				has_country_leader = {
					character = ITA_benito_mussolini
					ruling_only = yes
				}
				has_country_leader = {
					character = ITA_grand_council
					ruling_only = yes
				}
			}
		}
		option = {
			name = GermanyStealSouthTyrol.1.a
			ai_chance = {factor = 100}
			IF = {
				limit = {
					country_exists = AUS
				}
				AUS = {
					transfer_state = 39
					transfer_state = 850
					39 = { add_core_of = AUS }
					850 = { add_core_of = AUS }
				}
			}
			ELSE = {
				GER = {
					transfer_state = 39
					transfer_state = 850
					39 = { add_core_of = GER }
					850 = { add_core_of = GER }
				}
			}
			
		}
	}
